00039 /*
00040  * struct grouper_rule.op is a uint16_t with the options ORed together so that
00041  * one can have a switch over them.
00042  *
00043  * the comparison constants come first as they are not of the size of a power
00044  * of two and hence, would create wholes in the range of possible values for
00045  * grouper_rule.op which in turn would make it harder for the compiler to
00046  * optimize the switch. It is generally better if the possible values for the
00047  * switch statement are continuous because then, a branchtable can be built.
00048  *
00049  * +---+---+---+---+---+---+---+---+---+
00050  * |    comp   | delta | size1 | size2 |
00051  * +---+---+---+---+---+---+---+---+---+
00052  *
00053  * comp:  eq/ne/gt/lt/le/ge comparison operation
00054  * delta:     relative/absolute/no delta
00055  * size1: 8/16/32/64 bits for first argument
00056  * size2: 8/16/32/64 bits for second argument
00057  */
00058 
00059 enum {
00060     /* size2 */
00061     RULE_S2_8 = 0,
00062     RULE_S2_16 = 1,
00063     RULE_S2_32 = 2,
00064     RULE_S2_64 = 3,
00065     /* size1 */
00066     RULE_S1_8 = 0,
00067     RULE_S1_16 = 4,
00068     RULE_S1_32 = 8,
00069     RULE_S1_64 = 12,
00070     /* D */
00071     RULE_ABS = 0,
00072     RULE_REL = 16,
00073     RULE_NO = 32,
00074     /* comp */
00075     RULE_EQ = 0,
00076     RULE_NE = 64,
00077     RULE_GT = 128,
00078     RULE_LT = 192,
00079     RULE_LE = 256,
00080     RULE_GE = 320,
00081 };
